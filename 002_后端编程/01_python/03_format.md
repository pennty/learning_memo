```python
# <模板字符串>.format(<逗号分隔的参数>)

print("你{}, 世{} ".format("好", "界"))
print("{}{}{}".format("圆周率是", 3.1415926, "..."))
'''
你好, 世界
圆周率是 3.1415926...
'''

# 大括号输出, {: {{, }: }}

# 参数编号从 0 开始

print("圆周率{{{1}{2}}}是{0}".format("无理数", 3.1415926, "..."))
'''
圆周率{3.1415926...}是无理数
'''

# 格式控制信息

# {<参数序号>:<格式控制标记>}

# <格式控制标记>: <填充><对齐><宽度>,<.精度><类型>6 个字段

# 1. <填充>: 用于填充的单个字符

# 2. <对齐>: < 左对齐; > 右对齐; ^ 居中对齐

# 3. <宽度>: 槽的设定; 输入宽度

# 4. ,: 数字的千位分隔符, 适用于整数和浮点数

# 5. <.精度>: 浮点数小数部分的精度或字符串的最大输出长度

# 6. <类型>: 整数类型 B, c, d, o, x, X; 浮点数类型 e, E, f, %

s = "Python"
f = 123456.78901234

# 宽度

print("|{0:30}|".format(s))
'''
|Python |
'''

# 对齐

print("|{0:<30}|".format(s))
print("|{0:>30}|".format(s))
print("|{0:^30}|".format(s))
'''
|Python |
| Python|
| Python |
'''

# 填充

print("|{0:-<30}|".format(s))
print("|{0:->30}|".format(s))
print("|{0:-^30}|".format(s))
'''
|Python------------------------|
|------------------------Python|
|------------Python------------|
'''

# 逗号

print("|{0:^30,}|".format(f))
'''
| 123,456.78901234 |
'''

# .精度

print(("|{0:^30.4}|".format(f)))
print(("|{0:^30.4}|".format(s)))
'''
| 1.235e+05 |
| Pyth |
'''

# 类型

## 整数类型

## b: 二进制方式

## c: 对应的 Uncode 字符

## d: 十进制方式

## o: 八进制方式

## x: 小写十六进制方式

## X: 大写十六进制方式

print("{0:b},{0:c},{0:d},{0:o},{0:x},{0:X}".format(425))
'''
110101001,Ʃ,425,651,1a9,1A9
'''

## 浮点类型

## e: 输出浮点数对应的小写字母 e 的指数形式

## E: 输出浮点数对应的大写字母 E 的指数形式

## f: 输出浮点数的标准浮点形式

## %: 输出浮点数的百分形式

print("{0:e},{0:E},{0:f},{0:%}".format(3.14))
print("{0:.2e},{0:.2E},{0:.2f},{0:.2%}".format(3.14))
'''
3.140000e+00,3.140000E+00,3.140000,314.000000%
3.14e+00,3.14E+00,3.14,314.00%
'''
```
